\newproblem{04a}
{
	Solve the following equation for $y$. $$\frac{1}{4} y+5=  \frac{2}{3} y$$
}
{
	\begin{tabular}{l r}
	$12(\frac{1}{4}y+5)=  12(\frac{2}{3} y)$  & 1 pt to here\\
	$3y+60=8y$ & 2 pts to here\\
	$60=5y$ & 3 pts to here\\
	$y=12$  & 4 pts total
	\end{tabular}
}
