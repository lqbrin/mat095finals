\newproblem{04.5a}
{
	Solve the following equation for $y$. $$\frac{1}{3}(y-9)=  \frac{1}{6} y+2$$
}
{
	\begin{tabular}{l r}
	$\frac{1}{3}y-3=  \frac{1}{6} y+2$  & 1 pt to here\\
	$6\left(\frac{1}{3}y-3\right)=6 \left( \frac{1}{6} y+2\right)$  & 2 pts to here\\
	$2y-18=y+12$ & 3 pts to here\\
	$y=30$  & 4 pts total
	\end{tabular}
}
