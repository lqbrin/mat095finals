\newproblem{08.5a}
{
      Find the slope, $y$-intercept, and $x$-intercept of the line. $$7x-2y=4$$
}
{
	\begin{tabular}{l r}
	slope: $\dfrac{7}{2}$ & 1 pt \\
	$y$-intercept: $(0,-2)$ & 2 pts\\
	$x$-intercept: $\left(\dfrac{4}{7},0\right)$ & 2 pts\\
	\end{tabular}
}
