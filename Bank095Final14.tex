\newproblem{14a}
{
	\begin{enumerate}
	\item Rewrite without an exponent: $(-13)^{-2}$
	\begin{onlyproblem}\spc\end{onlyproblem}
	\item Rewrite without using a negative exponent: $-18x^{-6}$
	\end{enumerate}
}
{
	\begin{tabular}{l r}
	(a) $\dfrac{1}{169}$ & 2.5 pts\\
	& No partial credit.\\
	(b) $\dfrac{-18}{x^6}$ & 2.5 pts\\
	& No partial credit.
	\end{tabular}
}
