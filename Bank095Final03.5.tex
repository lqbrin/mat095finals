\newproblem{03.5a}
{
	Solve for $c$. Simplify answers. $$5(2c+3)-4=-5c+6$$
}
{
	\begin{tabular}{l r}
	$10c+15-4=-5c+6$ & 1 pt to here\\
	$10c+5c=-15+4+6$ & 2 pts to here\\
	$15c=-5$ & 3 pts to here\\
	$c=-\dfrac{5}{15}$ & 4 pts to here\\ \\
	$c=-\dfrac{1}{3}$ & 5 pts total\\
	\end{tabular}
}
