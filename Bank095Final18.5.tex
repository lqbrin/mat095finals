\newproblem{18.5a}
{
	Multiply and simplify $-7x^2y(-4x^2-6x+2y^3)$.
}
{
	\begin{tabular}{l r}
	$-7x^2y(-4x^2)-7x^2y(-6x)-7x^2y(2y^3)$ & 2 pts to here\\
	$=28x^4y+42x^3y-14x^2y^4$ & 1 pt for each correct term
	\end{tabular}
}
